   function sortMyString(s) {
     let res = "";
     let res1 = "";
     for (let index = 0; index &lt; s.length; index++) {
       if (index % 2 === 0) {
         res += s[index];
       } else {
         res1 += s[index];
       }
     }
     return res + " " + res1;
   }
